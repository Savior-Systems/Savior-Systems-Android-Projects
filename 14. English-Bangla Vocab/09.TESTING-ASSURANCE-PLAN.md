## 1. Automated Unit Test Cases (Lookup Logic)
`
```kotlin
package com.saviorsystems.englishbanglavocab
`
import com.saviorsystems.englishbanglavocab.data.database.entities.WordEntity
import org.junit.Assert.assertTrue
import org.junit.Test
`
class DictionarySearchTest {
`
    @Test
    fun matchesSearchPrefixCorrectly() {
        val word = WordEntity(english = "persistent", bangla = "অধ্যবসায়ী", ipa = null, synonyms = null)
`
        // Search query "pers" should prefix-match "persistent"
        val query = "pers"
        assertTrue(word.english.startsWith(query, ignoreCase = true))
    }
}
```
