## 3. Pacer ViewModel Implementation
`
```kotlin
package com.saviorsystems.breathingpacer.ui.main
`
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import kotlinx.coroutines.Job
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
`
data class PacerUiState(
    val breathingState: BreathingState = BreathingState.INHALE,
    val secondsRemainingInState: Int = 4,
    val isActive: Boolean = false
)
`
class PacerViewModel : ViewModel() {
`
    private val _uiState = MutableStateFlow(PacerUiState())
    val uiState: StateFlow<PacerUiState> = _uiState.asStateFlow()
`
    private var timerJob: Job? = null
`
    // Presets (Inhale, Hold, Exhale, Hold) in seconds
    private var currentPattern = intArrayOf(4, 4, 4, 4) 
`
    fun startSession() {
        timerJob?.cancel()
        _uiState.value = PacerUiState(isActive = true)
`
        timerJob = viewModelScope.launch {
            var stateIndex = 0
            val states = BreathingState.values()
`
            while (true) {
                val currentState = states[stateIndex]
                val duration = currentPattern[stateIndex]
`
                for (seconds in duration downTo 1) {
                    _uiState.value = _uiState.value.copy(
                        breathingState = currentState,
                        secondsRemainingInState = seconds
                    )
                    delay(1000L)
                }
`
                // Move to next state in cycle loop
                stateIndex = (stateIndex + 1) % states.size
            }
        }
    }
`
    fun stopSession() {
        timerJob?.cancel()
        _uiState.value = PacerUiState(isActive = false)
    }
`
    override fun onCleared() {
        super.onCleared()
        timerJob?.cancel()
    }
}
```
