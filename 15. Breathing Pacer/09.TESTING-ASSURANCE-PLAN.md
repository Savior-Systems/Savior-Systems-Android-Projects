## 1. Automated Unit Test Cases (State Sequencer)
`
```kotlin
package com.saviorsystems.breathingpacer
`
import com.saviorsystems.breathingpacer.ui.main.BreathingState
import org.junit.Assert.assertEquals
import org.junit.Test
`
class BreathingSequencerTest {
`
    @Test
    fun verifyBreathingStatesCycleCorrectly() {
        val states = BreathingState.values()
`
        // Cycle must be: INHALE -> HOLD_FULL -> EXHALE -> HOLD_EMPTY -> INHALE
        assertEquals(BreathingState.INHALE, states[0])
        assertEquals(BreathingState.HOLD_FULL, states[1])
        assertEquals(BreathingState.EXHALE, states[2])
        assertEquals(BreathingState.HOLD_EMPTY, states[3])
    }
}
```
