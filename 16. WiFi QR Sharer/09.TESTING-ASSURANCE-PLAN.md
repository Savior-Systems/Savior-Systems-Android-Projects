## 1. Automated Unit Test Cases (QR Formatting)
`
```kotlin
package com.saviorsystems.wifiqrsharer
`
import com.saviorsystems.wifiqrsharer.data.model.WifiNetwork
import com.saviorsystems.wifiqrsharer.utils.QrCodeGenerator
import org.junit.Assert.assertEquals
import org.junit.Test
`
class QrCodeGeneratorTest {
`
    @Test
    fun verifyWifiStringFormatting() {
        val network = WifiNetwork(
            ssid = "HomeNet",
            password = "password123",
            encryptionType = "WPA",
            isHidden = false
        )
`
        // Expected format: WIFI:S:HomeNet;T:WPA;P:password123;;
        val formatted = QrCodeGenerator.formatWifiString(network)
        assertEquals("WIFI:S:HomeNet;T:WPA;P:password123;;", formatted)
    }
}
```
