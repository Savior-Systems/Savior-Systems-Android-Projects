## 2. WiFi Network QR Code Formatting Utility
WIFI QR codes conform to the `ZXing` syntax. Below is the formatting logic:
```
```kotlin
package com.saviorsystems.wifiqrsharer.utils
```
import com.saviorsystems.wifiqrsharer.data.model.WifiNetwork
```
object QrCodeGenerator {
    /**
     * Formats network properties into standard WIFI string format.
     */
    fun formatWifiString(network: WifiNetwork): String {
        val escapedSsid = escapeString(network.ssid)
        val escapedPassword = escapeString(network.password)
        val hiddenSuffix = if (network.isHidden) "H:true;" else ""
```
        return "WIFI:S:$escapedSsid;T:${network.encryptionType};P:$escapedPassword;${hiddenSuffix};"
    }
```
    private fun escapeString(str: String): String {
        return str.replace("\\", "\\\\")
                  .replace(";", "\\;")
                  .replace(",", "\\,")
                  .replace(":", "\\:")
                  .replace("\"", "\\\"")
    }
}

## 3. Network Connection Manager (Android Q+)
To join a network programmatically under Android 10+ using `WifiNetworkSpecifier`:
```
```kotlin
package com.saviorsystems.wifiqrsharer.utils
```
import android.content.Context
import android.net.ConnectivityManager
import android.net.NetworkRequest
import android.net.NetworkSpecifier
import android.net.wifi.WifiNetworkSpecifier
import com.saviorsystems.wifiqrsharer.data.model.WifiNetwork
```
object WifiConnector {
    fun connectToWifi(context: Context, network: WifiNetwork) {
        val specifier = WifiNetworkSpecifier.Builder()
            .setSsid(network.ssid)
            .setWpa2Passphrase(network.password)
            .build()
```
        val request = NetworkRequest.Builder()
            .addTransportType(android.net.NetworkCapabilities.TRANSPORT_WIFI)
            .setNetworkSpecifier(specifier)
            .build()
```
        val connectivityManager = context.getSystemService(Context.CONNECTIVITY_SERVICE) as ConnectivityManager
        connectivityManager.requestNetwork(request, object : ConnectivityManager.NetworkCallback() {
            override fun onAvailable(network: android.net.Network) {
                super.onAvailable(network)
                connectivityManager.bindProcessToNetwork(network)
            }
        })
    }
}
