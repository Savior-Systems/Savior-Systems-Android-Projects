## 2. Core Date Math Engine (Kotlin)
`
This engine handles precise age calculation, including months, weeks, days, hours, and seconds, correctly adjusting for leap years.
`
```kotlin
package com.saviorsystems.smartage.engine
`
import java.time.LocalDate
import java.time.LocalDateTime
import java.time.Period
import java.time.temporal.ChronoUnit
`
object DateMathEngine {
`
    data class AgeResult(
        val years: Int,
        val months: Int,
        val days: Int,
        val totalMonths: Long,
        val totalWeeks: Long,
        val totalDays: Long,
        val totalHours: Long,
        val totalMinutes: Long,
        val totalSeconds: Long
    )
`
    fun calculateExactAge(birthDate: LocalDate, targetDate: LocalDate): AgeResult {
        require(!birthDate.isAfter(targetDate)) { "Birthdate cannot be after target date" }
`
        val period = Period.between(birthDate, targetDate)
`
        val birthDateTime = birthDate.atStartOfDay()
        val targetDateTime = targetDate.atStartOfDay()
`
        return AgeResult(
            years = period.years,
            months = period.months,
            days = period.days,
            totalMonths = ChronoUnit.MONTHS.between(birthDate, targetDate),
            totalWeeks = ChronoUnit.WEEKS.between(birthDate, targetDate),
            totalDays = ChronoUnit.DAYS.between(birthDate, targetDate),
            totalHours = ChronoUnit.HOURS.between(birthDateTime, targetDateTime),
            totalMinutes = ChronoUnit.MINUTES.between(birthDateTime, targetDateTime),
            totalSeconds = ChronoUnit.SECONDS.between(birthDateTime, targetDateTime)
        )
    }
`
    fun getDaysUntilNextBirthday(birthDate: LocalDate, targetDate: LocalDate): Long {
        val nextBD = birthDate.withYear(targetDate.year)
        val targetBD = if (nextBD.isBefore(targetDate) || nextBD.isEqual(targetDate)) {
            birthDate.withYear(targetDate.year + 1)
        } else {
            nextBD
        }
        return ChronoUnit.DAYS.between(targetDate, targetBD)
    }
}
```

## 3. AlarmManager Local Scheduler
`
*   **API Target**: Uses exact alarm schedules via `AlarmManager` for precise morning triggers.
*   **Safety Handling**: Checks `alarmManager.canScheduleExactAlarms()` on SDK 31+ to avoid crashing. If false, schedules inexact or falls back to a scheduled WorkManager periodic worker.
`
