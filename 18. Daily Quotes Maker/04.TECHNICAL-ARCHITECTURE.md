## 2. Graphic Canvas Rendering Logic (Export Utility)
To capture a Compose layout into a high-res bitmap offline on Android:
```
```kotlin
package com.saviorsystems.dailyquotesmaker.utils
```
import android.graphics.Bitmap
import android.graphics.Canvas
import android.view.View
```
object ImageExporter {
    /**
     * Converts a standard Android View (which hosts the Compose Canvas) into a Bitmap.
     */
    fun createBitmapFromView(view: View): Bitmap {
        val bitmap = Bitmap.createBitmap(
            view.width,
            view.height,
            Bitmap.Config.ARGB_8888
        )
        val canvas = Canvas(bitmap)
        view.draw(canvas)
        return bitmap
    }
}

## 3. Editor Active ViewModel
```
```kotlin
package com.saviorsystems.dailyquotesmaker.ui.editor
```
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.saviorsystems.dailyquotesmaker.data.database.entities.QuoteEntity
import com.saviorsystems.dailyquotesmaker.data.repository.QuoteRepository
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
```
data class EditorUiState(
    val currentText: String = "Design is thinking made visual.",
    val currentAuthor: String = "Saul Bass",
    val fontSizeSp: Float = 24f,
    val textColorHex: String = "#FFFFFF",
    val isSaved: Boolean = false
)
```
class EditorViewModel(
    private val repository: QuoteRepository
) : ViewModel() {
```
    private val _uiState = MutableStateFlow(EditorUiState())
    val uiState: StateFlow<EditorUiState> = _uiState.asStateFlow()
```
    fun updateText(text: String, author: String) {
        _uiState.update { it.copy(currentText = text, currentAuthor = author) }
    }
```
    fun updateStyle(fontSize: Float, textColor: String) {
        _uiState.update { it.copy(fontSizeSp = fontSize, textColorHex = textColor) }
    }
```
    fun saveQuoteToFavorites() {
        viewModelScope.launch {
            val entity = QuoteEntity(
                text = _uiState.value.currentText,
                author = _uiState.value.currentAuthor,
                timestamp = System.currentTimeMillis()
            )
            repository.insertFavorite(entity)
            _uiState.update { it.copy(isSaved = true) }
        }
    }
}
