# 10. Translation & Localization

## 1. Localization Strategy
The application supports English, Spanish, and Bengali. All labels, form headers, and system alerts are defined in localized XML resources.

---

## 2. Resource XML Files

### A. English (`values/strings.xml`)
```xml
<resources>
    <!-- Screen Titles & Index -->
    <string name="app_name">Resume Builder</string>
    <string name="title_dashboard">My Resumes</string>
    <string name="title_editor">Resume Editor</string>
    <string name="title_preview">Print Preview</string>

    <!-- Sections -->
    <string name="section_personal">Personal Details</string>
    <string name="section_education">Education</string>
    <string name="section_experience">Work Experience</string>
    <string name="section_projects">Projects</string>
    <string name="section_certifications">Certifications</string>
    <string name="section_skills">Skills</string>

    <!-- Customization Configs -->
    <string name="label_margins">Margins</string>
    <string name="margin_compact">Compact</string>
    <string name="margin_normal">Normal</string>
    <string name="margin_wide">Wide</string>
    <string name="label_fonts">Font Style</string>
    <string name="label_theme">Color Theme</string>

    <!-- Backup & Restore Actions -->
    <string name="action_export_json">Backup JSON</string>
    <string name="action_import_json">Import JSON</string>
    <string name="msg_backup_success">Backup successfully saved!</string>
    <string name="msg_import_success">Resume imported successfully!</string>

    <!-- PDF & Share Actions -->
    <string name="action_export_pdf">Export PDF</string>
    <string name="action_share">Share PDF</string>
    <string name="msg_pdf_success">PDF generated successfully!</string>

    <!-- Monetization & Unlock Alerts -->
    <string name="dialog_unlock_title">Unlock Premium Features</string>
    <string name="dialog_unlock_msg">Watch a short video ad to unlock professional themes, fonts, and margin controls for 24 hours.</string>
    <string name="action_watch_ad">Watch Ad</string>
    <string name="action_cancel">Cancel</string>
</resources>
```
