# 04. Technical Architecture Blueprint

## 1. Architecture Style: Clean MVVM (Model-View-ViewModel)
The application adheres to Clean Architecture layers with strict unidirectional data flow. The complex operations of measuring text layouts and compiling standard A4 documents are decoupled from Jetpack Compose components.

## 4. Kotlin Page Budgeting Algorithm
To prevent text from getting cut off on page boundaries (e.g., splitting a line of text or an experience item in half), the application utilizes a height-based layout budgeting algorithm:

```kotlin
package com.saviorsystems.resumemaker.domain.pdf

import android.graphics.Paint

data class PageLayout(val sections: MutableList<PrintableSection> = mutableListOf()) {
    fun isNotEmpty() = sections.isNotEmpty()
}

interface PrintableSection {
    fun measureHeight(paint: Paint, usableWidth: Float): Float
    fun canBeSplit(): Boolean
    fun splitAtHeight(remainingHeight: Float, paint: Paint): Pair<PrintableSection, PrintableSection>
}

class PageBudgetingEngine(
    val pageHeight: Float = 842f,       // A4 Height in points
    val pageWidth: Float = 595f,        // A4 Width in points
    val verticalMargin: Float = 54f,    // Default 0.75in margins
    val horizontalMargin: Float = 54f
) {
    fun partitionContent(
        sections: List<PrintableSection>,
        paint: Paint
    ): List<PageLayout> {
        val usableHeight = pageHeight - (verticalMargin * 2)
        val usableWidth = pageWidth - (horizontalMargin * 2)
        val pages = mutableListOf<PageLayout>()
        var currentPage = PageLayout()
        var currentHeight = 0f

        for (section in sections) {
            val sectionHeight = section.measureHeight(paint, usableWidth)

            if (currentHeight + sectionHeight <= usableHeight) {
                // Section fits on the current page
                currentPage.sections.add(section)
                currentHeight += sectionHeight
            } else {
                if (section.canBeSplit()) {
                    val remainingHeight = usableHeight - currentHeight
                    // Split the section (e.g. partition bullet points)
                    val (part1, part2) = section.splitAtHeight(remainingHeight, paint)
                    
                    currentPage.sections.add(part1)
                    pages.add(currentPage)

                    // Start a new page for the remaining part of the section
                    currentPage = PageLayout()
                    currentPage.sections.add(part2)
                    currentHeight = part2.measureHeight(paint, usableWidth)
                } else {
                    // Section cannot be split, push it entirely to a new page
                    pages.add(currentPage)
                    currentPage = PageLayout()
                    currentPage.sections.add(section)
                    currentHeight = sectionHeight
                }
            }
        }
        
        if (currentPage.isNotEmpty()) {
            pages.add(currentPage)
        }
        
        return pages
    }
}
```
* **Developer note:** Implement `PrintableSection` on elements like `ExperienceBlock` or `EducationBlock` to allow them to separate their bullet items dynamically into `part1` and `part2` when transitioning across pages.
