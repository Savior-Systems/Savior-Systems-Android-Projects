## 2. Closet Grid Filter ViewModel
`
```kotlin
package com.saviorsystems.outfitcanvas.ui.closet
`
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.saviorsystems.outfitcanvas.data.database.entities.ClothingItemEntity
import com.saviorsystems.outfitcanvas.data.repository.ClosetRepository
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
`
data class ClosetUiState(
    val items: List<ClothingItemEntity> = emptyList(),
    val selectedCategory: String = "ALL", // ALL, TOPS, BOTTOMS, SHOES
    val isLoading: Boolean = true
)
`
class ClosetViewModel(
    private val repository: ClosetRepository
) : ViewModel() {
`
    private val _selectedCategory = MutableStateFlow("ALL")
`
    val uiState: StateFlow<ClosetUiState> = combine(
        repository.getAllItems(),
        _selectedCategory
    ) { items, category ->
        val filtered = if (category == "ALL") items else items.filter { it.category.uppercase() == category }
        ClosetUiState(
            items = filtered,
            selectedCategory = category,
            isLoading = false
        )
    }.stateIn(
        scope = viewModelScope,
        started = SharingStarted.WhileSubscribed(5000),
        initialValue = ClosetUiState()
    )
`
    fun setCategoryFilter(category: String) {
        _selectedCategory.value = category
    }
`
    fun addClothingItem(imagePath: String, category: String, colorHex: String) {
        viewModelScope.launch {
            repository.insertItem(
                ClothingItemEntity(
                    imagePath = imagePath,
                    category = category,
                    colorHex = colorHex
                )
            )
        }
    }
}
```
