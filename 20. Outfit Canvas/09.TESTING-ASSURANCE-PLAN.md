```
## 1. Automated Unit Test Cases (Category Checks)
```
```kotlin
package com.saviorsystems.outfitcanvas
```
import com.saviorsystems.outfitcanvas.data.database.entities.ClothingItemEntity
import org.junit.Assert.assertTrue
import org.junit.Test
```
class ClosetCategoryTest {
```
    @Test
    fun verifyValidClosetCategory() {
        val item = ClothingItemEntity(imagePath = "path/to/img.jpg", category = "TOPS", colorHex = "#FFFFFF")
```
        // Category must match one of the allowed wardrobe catalog tags
        val validCategories = listOf("TOPS", "BOTTOMS", "SHOES", "OUTER")
        assertTrue(validCategories.contains(item.category.uppercase()))
    }
}
```
