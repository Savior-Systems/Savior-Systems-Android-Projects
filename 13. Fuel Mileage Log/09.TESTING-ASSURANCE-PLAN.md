## 1. Automated Unit Test Cases (Calculation Logic)
`
```kotlin
package com.saviorsystems.fuelmileagelog
`
import com.saviorsystems.fuelmileagelog.data.database.entities.FuelRecordEntity
import com.saviorsystems.fuelmileagelog.utils.MileageMath
import org.junit.Assert.assertEquals
import org.junit.Test
`
class MileageMathTest {
`
    @Test
    fun calculateMetricEfficiencyCorrectly() {
        val prev = FuelRecordEntity(odometer = 1000.0, volumeLiters = 0.0, totalCost = 0.0)
        val latest = FuelRecordEntity(odometer = 1500.0, volumeLiters = 35.0, totalCost = 50.0)
`
        // Efficiency = (35.0 / 500.0) * 100.0 = 7.0 L/100km
        val efficiency = MileageMath.calculateEfficiency(latest, prev, isMetric = true)
        assertEquals(7.0, efficiency, 0.001)
    }
`
    @Test
    fun calculateImperialEfficiencyCorrectly() {
        val prev = FuelRecordEntity(odometer = 1000.0, volumeLiters = 0.0, totalCost = 0.0)
        val latest = FuelRecordEntity(odometer = 1350.0, volumeLiters = 10.0, totalCost = 30.0)
`
        // Efficiency = 350.0 / 10.0 = 35.0 MPG
        val efficiency = MileageMath.calculateEfficiency(latest, prev, isMetric = false)
        assertEquals(35.0, efficiency, 0.001)
    }
}
```
