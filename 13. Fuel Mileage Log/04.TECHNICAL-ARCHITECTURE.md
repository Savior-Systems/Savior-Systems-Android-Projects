## 2. Fuel Mileage Calculation Logic
To calculate fuel consumption dynamically based on sequential odometer records:
`
```kotlin
package com.saviorsystems.fuelmileagelog.utils
`
import com.saviorsystems.fuelmileagelog.data.database.entities.FuelRecordEntity
`
object MileageMath {
    /**
     * Calculates fuel economy in L/100km or MPG based on the two latest logs.
     */
    fun calculateEfficiency(
        latestRecord: FuelRecordEntity,
        previousRecord: FuelRecordEntity,
        isMetric: Boolean
    ): Double {
        val distance = latestRecord.odometer - previousRecord.odometer
        if (distance <= 0) return 0.0
`
        return if (isMetric) {
            // Liters per 100 Kilometers
            (latestRecord.volumeLiters / distance) * 100.0
        } else {
            // Miles per Gallon (Assuming odometer is miles and volume is gallons)
            distance / latestRecord.volumeLiters
        }
    }
}
```

## 3. Dashboard ViewModel Implementation
`
```kotlin
package com.saviorsystems.fuelmileagelog.ui.main
`
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.saviorsystems.fuelmileagelog.data.database.entities.FuelRecordEntity
import com.saviorsystems.fuelmileagelog.data.repository.FuelRepository
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
`
data class FuelUiState(
    val records: List<FuelRecordEntity> = emptyList(),
    val isMetric: Boolean = true,
    val averageConsumption: Double = 0.0
)
`
class FuelViewModel(
    private val repository: FuelRepository
) : ViewModel() {
`
    private val _isMetric = MutableStateFlow(true)
`
    val uiState: StateFlow<FuelUiState> = combine(
        repository.getAllRecords(),
        _isMetric
    ) { records, isMetric ->
        var totalConsumption = 0.0
        var count = 0
`
        // Calculate dynamic averages based on adjacent sorted logs
        for (i in 0 until records.size - 1) {
            val eff = com.saviorsystems.fuelmileagelog.utils.MileageMath.calculateEfficiency(
                records[i],
                records[i + 1],
                isMetric
            )
            if (eff > 0.0) {
                totalConsumption += eff
                count++
            }
        }
`
        FuelUiState(
            records = records,
            isMetric = isMetric,
            averageConsumption = if (count > 0) totalConsumption / count else 0.0
        )
    }.stateIn(
        scope = viewModelScope,
        started = SharingStarted.WhileSubscribed(5000),
        initialValue = FuelUiState()
    )
`
    fun addFuelRecord(odometer: Double, volume: Double, cost: Double) {
        viewModelScope.launch {
            repository.insertRecord(
                FuelRecordEntity(
                    odometer = odometer,
                    volumeLiters = volume,
                    totalCost = cost,
                    timestamp = System.currentTimeMillis()
                )
            )
        }
    }
}
```
