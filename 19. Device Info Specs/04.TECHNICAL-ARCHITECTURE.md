## 2. RAM Memory Reader Logic
Below is the clean implementation to read memory metrics using Android's `ActivityManager`:
```
```kotlin
package com.saviorsystems.deviceinfospecs.data.system
```
import android.app.ActivityManager
import android.content.Context
```
object MemoryReader {
    data class MemoryMetrics(
        val totalBytes: Long,
        val availableBytes: Long
    )
```
    /**
     * Reads the current system memory metrics.
     */
    fun readMemory(context: Context): MemoryMetrics {
        val activityManager = context.getSystemService(Context.ACTIVITY_SERVICE) as ActivityManager
        val memoryInfo = ActivityManager.MemoryInfo()
        activityManager.getMemoryInfo(memoryInfo)
```
        return MemoryMetrics(
            totalBytes = memoryInfo.totalMem,
            availableBytes = memoryInfo.availMem
        )
    }
}

## 3. Real-time Specs ViewModel
```
```kotlin
package com.saviorsystems.deviceinfospecs.ui.main
```
import android.content.Context
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.saviorsystems.deviceinfospecs.data.system.MemoryReader
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
```
data class SpecsUiState(
    val totalMemoryBytes: Long = 0L,
    val availableMemoryBytes: Long = 0L,
    val batteryPct: Float = 0f,
    val triggerTick: Long = 0L
)
```
class SpecsViewModel(
    private val context: Context
) : ViewModel() {
```
    private val _uiState = MutableStateFlow(SpecsUiState())
    val uiState: StateFlow<SpecsUiState> = _uiState.asStateFlow()
```
    init {
        // Start live update loop for RAM and CPU monitor tick
        viewModelScope.launch {
            while (true) {
                val mem = MemoryReader.readMemory(context)
                _uiState.update { 
                    it.copy(
                        totalMemoryBytes = mem.totalBytes,
                        availableMemoryBytes = mem.availableBytes,
                        triggerTick = System.currentTimeMillis()
                    )
                }
                delay(2000L) // Refresh every 2 seconds
            }
        }
    }
}
