## 1. Room Database: `AppDatabase.kt`

```kotlin
@Database(
    entities = [FocusSession::class, WeeklyChallenge::class, StreakShield::class],
    version = 2,
    exportSchema = true
)
@TypeConverters(Converters::class)
abstract class AppDatabase : RoomDatabase() {
    abstract fun sessionDao(): SessionDao
    abstract fun challengeDao(): ChallengeDao
    abstract fun shieldDao(): ShieldDao

    companion object {
        val MIGRATION_1_2 = object : Migration(1, 2) {
            override fun migrate(db: SupportSQLiteDatabase) {
                db.execSQL("""
                    CREATE TABLE IF NOT EXISTS weekly_challenge (
                        id INTEGER PRIMARY KEY AUTOINCREMENT,
                        start_date INTEGER NOT NULL,
                        target_sessions INTEGER NOT NULL,
                        completed_sessions INTEGER NOT NULL DEFAULT 0,
                        reward_type TEXT NOT NULL DEFAULT 'SHIELD',
                        is_complete INTEGER NOT NULL DEFAULT 0
                    )
                """)
                db.execSQL("""
                    CREATE TABLE IF NOT EXISTS streak_shield (
                        id INTEGER PRIMARY KEY DEFAULT 1,
                        shield_count INTEGER NOT NULL DEFAULT 0,
                        last_gift_date INTEGER NOT NULL DEFAULT 0,
                        last_rewarded_date INTEGER NOT NULL DEFAULT 0
                    )
                """)
            }
        }
    }
}
```

## 2. Entities

### Entity: `focus_session`

```kotlin
@Entity(tableName = "focus_session")
data class FocusSession(
    @PrimaryKey(autoGenerate = true) val id: Int = 0,
    @ColumnInfo(name = "timestamp") val timestamp: Long,       // epoch ms of completion
    @ColumnInfo(name = "duration_minutes") val durationMinutes: Int,
    @ColumnInfo(name = "session_type") val sessionType: String  // "WORK" | "SHORT_BREAK" | "LONG_BREAK"
)
```

### Entity: `weekly_challenge` 🆕

```kotlin
@Entity(tableName = "weekly_challenge")
data class WeeklyChallenge(
    @PrimaryKey(autoGenerate = true) val id: Int = 0,
    @ColumnInfo(name = "start_date") val startDate: Long,           // Monday 00:00 epoch ms
    @ColumnInfo(name = "target_sessions") val targetSessions: Int,  // 10 / 20 / 30
    @ColumnInfo(name = "completed_sessions") val completedSessions: Int = 0,
    @ColumnInfo(name = "reward_type") val rewardType: String = "SHIELD", // "SHIELD" | "CARD_THEME"
    @ColumnInfo(name = "is_complete") val isComplete: Boolean = false
)
```

### Entity: `streak_shield` 🆕

```kotlin
@Entity(tableName = "streak_shield")
data class StreakShield(
    @PrimaryKey val id: Int = 1,                          // Singleton row
    @ColumnInfo(name = "shield_count") val shieldCount: Int = 0,            // 0, 1, or 2 max
    @ColumnInfo(name = "last_gift_date") val lastGiftDate: Long = 0L,        // epoch ms of last daily gift
    @ColumnInfo(name = "last_rewarded_date") val lastRewardedDate: Long = 0L // epoch ms of last rewarded grant
)
```

## 3. DAO Signatures

### `SessionDao.kt`

```kotlin
@Dao
interface SessionDao {

    @Insert
    suspend fun insertSession(session: FocusSession)

    @Query("SELECT * FROM focus_session ORDER BY timestamp DESC")
    fun getAllSessions(): Flow<List<FocusSession>>

    @Query("SELECT * FROM focus_session WHERE timestamp >= :startMs ORDER BY timestamp DESC")
    fun getSessionsSince(startMs: Long): Flow<List<FocusSession>>

    // For heatmap: count WORK sessions grouped by calendar day
    @Query("""
        SELECT date(timestamp / 1000, 'unixepoch', 'localtime') as day, COUNT(*) as count
        FROM focus_session
        WHERE session_type = 'WORK' AND timestamp >= :startMs
        GROUP BY day ORDER BY day ASC
    """)
    suspend fun getDailySessionCounts(startMs: Long): List<DaySessions>  // DaySessions(day: String, count: Int)

    // For Smart Goal Adaptation: average WORK sessions per day over last 7 days
    @Query("""
        SELECT AVG(daily_count) FROM (
            SELECT COUNT(*) as daily_count
            FROM focus_session
            WHERE session_type = 'WORK' AND timestamp >= :sevenDaysAgo
            GROUP BY date(timestamp / 1000, 'unixepoch', 'localtime')
        )
    """)
    suspend fun getAverageSessionsPerDay(sevenDaysAgo: Long): Float

    // For day-of-week insight: "You focus best on Tuesdays"
    @Query("""
        SELECT strftime('%w', timestamp / 1000, 'unixepoch', 'localtime') as dow, COUNT(*) as count
        FROM focus_session WHERE session_type = 'WORK'
        GROUP BY dow ORDER BY count DESC LIMIT 1
    """)
    suspend fun getBestDayOfWeek(): DayOfWeekResult  // DayOfWeekResult(dow: String, count: Int)

    // For weekly challenge: count WORK sessions in a date range
    @Query("SELECT COUNT(*) FROM focus_session WHERE session_type = 'WORK' AND timestamp BETWEEN :startMs AND :endMs")
    suspend fun countSessionsInRange(startMs: Long, endMs: Long): Int

    // For today's session count
    @Query("""
        SELECT COUNT(*) FROM focus_session
        WHERE session_type = 'WORK'
        AND date(timestamp / 1000, 'unixepoch', 'localtime') = date('now', 'localtime')
    """)
    fun getTodaySessionCount(): Flow<Int>

    @Query("DELETE FROM focus_session")
    suspend fun clearHistory()
}
```

### `ShieldDao.kt` 🆕

```kotlin
@Dao
interface ShieldDao {

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertOrUpdate(shield: StreakShield)

    @Query("SELECT * FROM streak_shield WHERE id = 1")
    suspend fun getShield(): StreakShield?

    @Query("UPDATE streak_shield SET shield_count = MAX(0, shield_count - 1) WHERE id = 1")
    suspend fun consumeShield()

    @Query("UPDATE streak_shield SET shield_count = MIN(2, shield_count + 1), last_rewarded_date = :timestamp WHERE id = 1")
    suspend fun addRewardedShield(timestamp: Long)
}
```
