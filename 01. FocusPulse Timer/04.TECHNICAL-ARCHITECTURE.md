```kotlin
    fun shareCard(activity: android.app.Activity, bitmap: android.graphics.Bitmap) {
        val file = java.io.File(activity.cacheDir, "focus_card.png")
        file.outputStream().use { bitmap.compress(android.graphics.Bitmap.CompressFormat.PNG, 100, it) }
        val uri = androidx.core.content.FileProvider.getUriForFile(

            putExtra(android.content.Intent.EXTRA_STREAM, uri)
            addFlags(android.content.Intent.FLAG_GRANT_READ_URI_PERMISSION)
        }
        activity.startActivity(android.content.Intent.createChooser(intent, "Share your focus 🔥"))
    }
```

```kotlin
class FocusWidget : GlanceAppWidget() {

    override suspend fun provideGlance(context: Context, id: GlanceId) {

                        .fillMaxWidth()
                        .background(Color(0xFF1A252F))

                    // Circular ring placeholder (Glance does not support Canvas — use Image drawable)
```

## 5. 🆕 AppOpenAdManager.kt

```kotlin
object AppOpenAdManager : Application.ActivityLifecycleCallbacks {

    private var appOpenAd: AppOpenAd? = null
    private var lastAdShowTime: Long = 0L
    private const val AD_COOLDOWN_MS = 60_000L  // 60 seconds

    fun loadAd(context: Context, adUnitId: String) {
        AppOpenAd.load(
            context, adUnitId, AdRequest.Builder().build(),
            object : AppOpenAd.AppOpenAdLoadCallback() {
                override fun onAdLoaded(ad: AppOpenAd) { appOpenAd = ad }
                override fun onAdFailedToLoad(error: LoadAdError) { appOpenAd = null }
            }
        )
    }

    fun showIfAvailable(activity: Activity, onComplete: () -> Unit) {
        val elapsed = System.currentTimeMillis() - lastAdShowTime
        if (appOpenAd != null && elapsed >= AD_COOLDOWN_MS) {
            appOpenAd!!.fullScreenContentCallback = object : FullScreenContentCallback() {
                override fun onAdDismissedFullScreenContent() {
                    appOpenAd = null
                    lastAdShowTime = System.currentTimeMillis()
                    onComplete()
                }
                override fun onAdFailedToShowFullScreenContent(e: AdError) { onComplete() }
            }
            appOpenAd!!.show(activity)
        } else {
            onComplete()
        }
    }
}
```

## 6. 🆕 ChallengeRepository.kt — Weekly Challenge Logic

```kotlin
class ChallengeRepository(
    private val challengeDao: ChallengeDao,
    private val sessionDao: SessionDao
) {
    suspend fun checkAndResetWeekly(): WeeklyChallenge {
        val existing = challengeDao.getCurrentChallenge()
        val thisMonday = getThisMondayEpoch()

        return if (existing == null || existing.startDate < thisMonday) {
            val newTarget = determineTargetForUser()
            val challenge = WeeklyChallenge(
                startDate = thisMonday,
                targetSessions = newTarget,
                completedSessions = 0,
                isComplete = false
            )
            challengeDao.insert(challenge)
            challenge
        } else {
            existing
        }
    }

    suspend fun onSessionCompleted(): Boolean {
        val challenge = challengeDao.getCurrentChallenge() ?: return false
        if (challenge.isComplete) return false

        val updated = challenge.copy(completedSessions = challenge.completedSessions + 1)
        if (updated.completedSessions >= updated.targetSessions) {
            challengeDao.update(updated.copy(isComplete = true))
            return true  // challenge just completed — caller shows celebration
        }
        challengeDao.update(updated)
        return false
    }

    private fun getThisMondayEpoch(): Long {
        val cal = java.util.Calendar.getInstance()
        cal.set(java.util.Calendar.DAY_OF_WEEK, java.util.Calendar.MONDAY)
        cal.set(java.util.Calendar.HOUR_OF_DAY, 0); cal.set(java.util.Calendar.MINUTE, 0)
        cal.set(java.util.Calendar.SECOND, 0); cal.set(java.util.Calendar.MILLISECOND, 0)
        return cal.timeInMillis
    }

    private suspend fun determineTargetForUser(): Int {
        val completedLastWeek = sessionDao.countSessionsInRange(
            System.currentTimeMillis() - 7 * 86_400_000L,
            System.currentTimeMillis()
        )
        return when {
            completedLastWeek >= 25 -> 30  // Advanced
            completedLastWeek >= 15 -> 20  // Standard
            else -> 10                      // Beginner
        }
    }
}
```
