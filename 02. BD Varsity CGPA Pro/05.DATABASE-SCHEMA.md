# 05. Database Schema & Storage
`
## 1. Room Database Overview
`
*   **Database Class**: `BdCgpaDatabase`
*   **Database Name**: `bd_cgpa_db`
*   **Version**: 1
*   **Pre-populated**: Yes — university grading scales loaded from `assets/databases/universities.db`
*   **Export Schema**: `true` (for migration tracking)
`
```kotlin
@Database(
    entities = [
        UniversityEntity::class,
        SemesterEntity::class,
        CourseEntity::class,
        HistoryEntity::class,
    ],
    version = 1,
    exportSchema = true,
)
@TypeConverters(GradeScaleConverter::class)
abstract class BdCgpaDatabase : RoomDatabase() {
    abstract fun universityDao(): UniversityDao
    abstract fun semesterDao(): SemesterDao
    abstract fun courseDao(): CourseDao
    abstract fun historyDao(): HistoryDao
}
```

## 2. Entity Definitions
`
### 2.1 University Table (`university_table`)
`
Stores both pre-loaded and user-created custom university grading scales.
`
```kotlin
@Entity(tableName = "university_table")
data class UniversityEntity(
    @PrimaryKey(autoGenerate = true)
    val id: Int = 0,
    val name: String,                    // "Dhaka University (DU)"
    val nameBn: String,                  // "ঢাকা বিশ্ববিদ্যালয় (ঢাবি)"
    val shortName: String,               // "DU"
    val type: String,                    // "PUBLIC", "PRIVATE", "CUSTOM"
    val isCustom: Boolean,               // false for presets, true for user-created
    val gradingScaleJson: String,        // JSON array of GradeEntry objects
)
```
`
| Column | Type | Constraints | Description |
|:-------|:-----|:------------|:------------|
| `id` | INTEGER | PK, AutoGenerate | Unique identifier |
| `name` | TEXT | NOT NULL | Full English name |
| `name_bn` | TEXT | NOT NULL | Full Bangla name |
| `short_name` | TEXT | NOT NULL | Abbreviated name (DU, NU, BUET) |
| `type` | TEXT | NOT NULL | PUBLIC, PRIVATE, or CUSTOM |
| `is_custom` | INTEGER (BOOLEAN) | NOT NULL | 0 = preset, 1 = user-created |
| `grading_scale_json` | TEXT | NOT NULL | Serialized `List<GradeEntry>` |
`
### 2.2 Semester Table (`semester_table`)
`
```kotlin
@Entity(
    tableName = "semester_table",
    foreignKeys = [
        ForeignKey(
            entity = UniversityEntity::class,
            parentColumns = ["id"],
            childColumns = ["universityId"],
            onDelete = ForeignKey.CASCADE,
        )
    ],
    indices = [Index("universityId")]
)
data class SemesterEntity(
    @PrimaryKey(autoGenerate = true)
    val id: Long = 0,
    val universityId: Int,               // FK → university_table.id
    val name: String,                    // "Semester 1"
    val orderIndex: Int,                 // Display order (0-based)
    val createdAt: Long,                 // System.currentTimeMillis()
)
```
`
| Column | Type | Constraints | Description |
|:-------|:-----|:------------|:------------|
| `id` | INTEGER | PK, AutoGenerate | Unique identifier |
| `university_id` | INTEGER | FK → university_table, CASCADE | Owning university |
| `name` | TEXT | NOT NULL | User-visible semester label |
| `order_index` | INTEGER | NOT NULL | Sort order |
| `created_at` | INTEGER | NOT NULL | Creation timestamp (epoch ms) |
`
### 2.3 Course Table (`course_table`)
`
```kotlin
@Entity(
    tableName = "course_table",
    foreignKeys = [
        ForeignKey(
            entity = SemesterEntity::class,
            parentColumns = ["id"],
            childColumns = ["semesterId"],
            onDelete = ForeignKey.CASCADE,
        )
    ],
    indices = [Index("semesterId")]
)
data class CourseEntity(
    @PrimaryKey(autoGenerate = true)
    val id: Long = 0,
    val semesterId: Long,                // FK → semester_table.id
    val name: String,                    // "Data Structures" or ""
    val creditHours: Double,             // 3.0
    val letterGrade: String?,            // "A+" or null
    val gradePoint: Double?,             // 4.00 or null
)
```
`
| Column | Type | Constraints | Description |
|:-------|:-----|:------------|:------------|
| `id` | INTEGER | PK, AutoGenerate | Unique identifier |
| `semester_id` | INTEGER | FK → semester_table, CASCADE | Owning semester |
| `name` | TEXT | NOT NULL (can be empty) | Course title |
| `credit_hours` | REAL | NOT NULL | Credit weight (0.5 - 6.0) |
| `letter_grade` | TEXT | NULLABLE | Selected letter grade |
| `grade_point` | REAL | NULLABLE | Derived grade point value |
`
### 2.4 Calculation History Table (`history_table`)
`
```kotlin
@Entity(tableName = "history_table")
data class HistoryEntity(
    @PrimaryKey(autoGenerate = true)
    val id: Long = 0,
    val universityName: String,          // "Dhaka University (DU)"
    val universityShortName: String,     // "DU"
    val semesterCount: Int,              // 6
    val totalCredits: Double,            // 120.0
    val totalCourses: Int,               // 42
    val cumulativeCgpa: Double,          // 3.67
    val timestamp: Long,                 // System.currentTimeMillis()
)
```
`
| Column | Type | Constraints | Description |
|:-------|:-----|:------------|:------------|
| `id` | INTEGER | PK, AutoGenerate | Unique identifier |
| `university_name` | TEXT | NOT NULL | University at time of save |
| `university_short_name` | TEXT | NOT NULL | Short name for display |
| `semester_count` | INTEGER | NOT NULL | Number of semesters |
| `total_credits` | REAL | NOT NULL | Sum of all credit hours |
| `total_courses` | INTEGER | NOT NULL | Number of graded courses |
| `cumulative_cgpa` | REAL | NOT NULL | Final CGPA result |
| `timestamp` | INTEGER | NOT NULL | Save time (epoch ms) |

## 3. DAO Interfaces
`
### 3.1 UniversityDao
`
```kotlin
@Dao
interface UniversityDao {
    @Query("SELECT * FROM university_table ORDER BY type ASC, name ASC")
    fun getAllUniversities(): Flow<List<UniversityEntity>>
`
    @Query("SELECT * FROM university_table WHERE id = :id")
    suspend fun getUniversityById(id: Int): UniversityEntity?
`
    @Query("SELECT * FROM university_table WHERE type = :type ORDER BY name ASC")
    fun getUniversitiesByType(type: String): Flow<List<UniversityEntity>>
`
    @Query("SELECT * FROM university_table WHERE name LIKE '%' || :query || '%' OR short_name LIKE '%' || :query || '%' OR name_bn LIKE '%' || :query || '%'")
    fun searchUniversities(query: String): Flow<List<UniversityEntity>>
`
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertUniversity(university: UniversityEntity): Long
`
    @Update
    suspend fun updateUniversity(university: UniversityEntity)
`
    @Delete
    suspend fun deleteUniversity(university: UniversityEntity)
`
    @Query("DELETE FROM university_table WHERE is_custom = 1")
    suspend fun deleteAllCustomUniversities()
}
```

## 4. Type Converters
`
```kotlin
class GradeScaleConverter {
    private val gson = Gson()
    private val type = object : TypeToken<List<GradeEntryJson>>() {}.type
`
    @TypeConverter
    fun fromGradeScale(gradeScale: String): List<GradeEntryJson> {
        return gson.fromJson(gradeScale, type) ?: emptyList()
    }
`
    @TypeConverter
    fun toGradeScale(gradeScale: List<GradeEntryJson>): String {
        return gson.toJson(gradeScale, type)
    }
}
`
data class GradeEntryJson(
    val letter: String,
    val point: Double,
    val minPct: Int?,
    val maxPct: Int?,
)
```

## 6. DataStore Preferences (Jetpack DataStore)
`
Stores user settings that persist across sessions.
`
```kotlin
object PreferenceKeys {
    val SELECTED_UNIVERSITY_ID = intPreferencesKey("selected_university_id")  // Default: 1 (DU)
    val THEME_MODE = intPreferencesKey("theme_mode")                          // 0=System, 1=Light, 2=Dark
    val LANGUAGE = stringPreferencesKey("language")                           // "en" or "bn"
    val IS_FIRST_LAUNCH = booleanPreferencesKey("is_first_launch")            // Default: true
}
```
`
| Key | Type | Default | Description |
|:----|:-----|:--------|:------------|
| `selected_university_id` | Int | 1 | Currently active university ID |
| `theme_mode` | Int | 0 | 0 = Follow System, 1 = Light, 2 = Dark |
| `language` | String | "en" | App display language ("en" or "bn") |
| `is_first_launch` | Boolean | true | Triggers university selection screen |
