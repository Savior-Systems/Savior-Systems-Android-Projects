## 2. Countdown Math (Duration Calculator)
```
```kotlin
package com.saviorsystems.examcountdownbd.utils
```
data class TimeRemaining(
    val days: Long,
    val hours: Long,
    val minutes: Long,
    val seconds: Long
)
```
object CountdownMath {
    /**
     * Calculates time details remaining between now and target timestamp.
     */
    fun calculateTimeRemaining(targetTimeMs: Long): TimeRemaining {
        val diffMs = targetTimeMs - System.currentTimeMillis()
        if (diffMs <= 0) return TimeRemaining(0, 0, 0, 0)
```
        val totalSecs = diffMs / 1000
        val days = totalSecs / 86400
        val remainderDays = totalSecs % 86400
        val hours = remainderDays / 3600
        val remainderHours = remainderDays % 3600
        val minutes = remainderHours / 60
        val seconds = remainderHours % 60
```
        return TimeRemaining(days, hours, minutes, seconds)
    }
}

## 3. Active Dashboard ViewModel
```
```kotlin
package com.saviorsystems.examcountdownbd.ui.main
```
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.saviorsystems.examcountdownbd.data.database.entities.ExamEntity
import com.saviorsystems.examcountdownbd.data.repository.ExamRepository
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
```
data class ExamUiState(
    val exams: List<ExamEntity> = emptyList(),
    val triggerTick: Long = 0L // Updates every second to trigger compose redraws
)
```
class ExamViewModel(
    private val repository: ExamRepository
) : ViewModel() {
```
    private val _tick = MutableStateFlow(0L)
```
    val uiState: StateFlow<ExamUiState> = combine(
        repository.getAllExams(),
        _tick
    ) { exams, tick ->
        ExamUiState(exams = exams, triggerTick = tick)
    }.stateIn(
        scope = viewModelScope,
        started = SharingStarted.WhileSubscribed(5000),
        initialValue = ExamUiState()
    )
```
    init {
        // Start countdown tick loop
        viewModelScope.launch {
            while (true) {
                delay(1000L)
                _tick.value = System.currentTimeMillis()
            }
        }
    }
```
    fun addExam(title: String, targetTimestamp: Long, category: String) {
        viewModelScope.launch {
            repository.insertExam(
                ExamEntity(
                    title = title,
                    targetTimestamp = targetTimestamp,
                    category = category
                )
            )
        }
    }
}
