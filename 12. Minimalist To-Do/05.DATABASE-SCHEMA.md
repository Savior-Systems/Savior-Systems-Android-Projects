# 05. Database Schema
`
This document details the SQLite/Room database schema, relationship constraints, indexes, and DAO queries for the **Minimalist To-Do** app.
`

`
## 1. Entity Diagrams

│          categories                  │          │               tasks                  │
├──────────────────────────────────────┤          ├──────────────────────────────────────┤
│ PK  id          INTEGER (AUTOINCREMENT)│ <─────── │ FK  category_id INTEGER (ON DELETE CASCADE)│

## 2. Table Definitions
`
### `categories` Table
Stores custom groupings created by the user (e.g., "Work", "Personal").
*   `id`: `INTEGER` (Primary Key, Autoincrement)
*   `name`: `TEXT` (Not Null)
*   `color_hex`: `TEXT` (Not Null, default value `#4DB6AC`)
`
### `tasks` Table
Stores the user's checklist items.
*   `id`: `INTEGER` (Primary Key, Autoincrement)
*   `title`: `TEXT` (Not Null)
*   `priority`: `INTEGER` (Not Null, default `1` (Medium))
*   `due_date`: `INTEGER` (Nullable Milliseconds timestamp)
*   `is_completed`: `INTEGER` (Not Null, default `0` (False))
*   `created_at`: `INTEGER` (Not Null, Milliseconds timestamp)
*   `category_id`: `INTEGER` (Not Null, Foreign Key referencing `categories(id)` ON DELETE CASCADE)
`

## 3. Room Entities Implementation
`
```kotlin
package com.saviorsystems.minimalisttodo.data.database.entities
`
import androidx.room.ColumnInfo
import androidx.room.Entity
import androidx.room.ForeignKey
import androidx.room.Index
import androidx.room.PrimaryKey
`
@Entity(tableName = "categories")
data class CategoryEntity(
    @PrimaryKey(autoGenerate = true) val id: Long = 0L,
    val name: String,
    @ColumnInfo(name = "color_hex") val colorHex: String = "#4DB6AC"
)
`
@Entity(
    tableName = "tasks",
    foreignKeys = [
        ForeignKey(
            entity = CategoryEntity::class,
            parentColumns = ["id"],
            childColumns = ["category_id"],
            onDelete = ForeignKey.CASCADE
        )
    ],
    indices = [
        Index(value = ["category_id"]),
        Index(value = ["is_completed", "created_at"])
    ]
)
data class TaskEntity(
    @PrimaryKey(autoGenerate = true) val id: Long = 0L,
    val title: String,
    val priority: Int = 1, // 0 = Low, 1 = Medium, 2 = High
    @ColumnInfo(name = "due_date") val dueDate: Long? = null,
    @ColumnInfo(name = "is_completed") val isCompleted: Boolean = false,
    @ColumnInfo(name = "created_at") val createdAt: Long = System.currentTimeMillis(),
    @ColumnInfo(name = "category_id") val categoryId: Long
)
```

## 4. Room DAO (Data Access Object)
`
```kotlin
package com.saviorsystems.minimalisttodo.data.database
`
import androidx.room.*
import com.saviorsystems.minimalisttodo.data.database.entities.CategoryEntity
import com.saviorsystems.minimalisttodo.data.database.entities.TaskEntity
import kotlinx.coroutines.flow.Flow
`
@Dao
interface TaskDao {
`
    // --- CATEGORIES ---
    @Query("SELECT * FROM categories ORDER BY id ASC")
    fun getAllCategories(): Flow<List<CategoryEntity>>
`
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertCategory(category: CategoryEntity): Long
`
    @Delete
    suspend fun deleteCategory(category: CategoryEntity)
`
    // --- TASKS ---
    @Query("SELECT * FROM tasks WHERE is_completed = 0 ORDER BY created_at DESC")
    fun getAllActiveTasks(): Flow<List<TaskEntity>>
`
    @Query("SELECT * FROM tasks WHERE category_id = :categoryId AND is_completed = 0 ORDER BY created_at DESC")
    fun getTasksByCategory(categoryId: Long): Flow<List<TaskEntity>>
`
    @Query("SELECT * FROM tasks WHERE is_completed = 1 ORDER BY created_at DESC")
    fun getCompletedTasks(): Flow<List<TaskEntity>>
`
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertTask(task: TaskEntity): Long
`
    @Update
    suspend fun updateTask(task: TaskEntity)
`
    @Delete
    suspend fun deleteTask(task: TaskEntity)
`
    @Query("DELETE FROM tasks WHERE is_completed = 1")
    suspend fun clearCompletedTasks()
}
```

## 5. Migration Strategy
*   **Database Version**: `1` (Initial Release).
*   **Future Upgrades**: Any schema changes (e.g., adding tags or voice memos) will require manual migrations via Room's `Migration` API rather than destructive fallback rebuilds, protecting offline user tasks.
