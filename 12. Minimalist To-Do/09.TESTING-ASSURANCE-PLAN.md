## 1. Automated Unit Test Cases (Room & ViewModel)
`
Below is the Kotlin implementation for testing Room task queries, sorting preferences, and ViewModel emissions:
`
```kotlin
package com.saviorsystems.minimalisttodo
`
import android.content.Context
import androidx.room.Room
import androidx.test.core.app.ApplicationProvider
import androidx.test.ext.junit.runners.AndroidJUnit4
import com.saviorsystems.minimalisttodo.data.database.TaskDao
import com.saviorsystems.minimalisttodo.data.database.TaskDatabase
import com.saviorsystems.minimalisttodo.data.database.entities.CategoryEntity
import com.saviorsystems.minimalisttodo.data.database.entities.TaskEntity
import kotlinx.coroutines.flow.first
import kotlinx.coroutines.runBlocking
import org.junit.After
import org.junit.Assert.assertEquals
import org.junit.Before
import org.junit.Test
import org.junit.runner.RunWith
import java.io.IOException
`
@RunWith(AndroidJUnit4::class)
class TaskDatabaseTest {
    private lateinit var db: TaskDatabase
    private lateinit var dao: TaskDao
`
    @Before
    fun createDb() {
        val context = ApplicationProvider.getApplicationContext<Context>()
        db = Room.inMemoryDatabaseBuilder(context, TaskDatabase::class.java).build()
        dao = db.taskDao()
    }
`
    @After
    @Throws(IOException::class)
    fun closeDb() {
        db.close()
    }
`
    @Test
    @Throws(Exception::class)
    fun writeTaskAndReadInList() = runBlocking {
        // Insert category
        val categoryId = dao.insertCategory(CategoryEntity(name = "Work"))
`
        // Insert task
        val task = TaskEntity(
            title = "Write QA Plan",
            categoryId = categoryId,
            isCompleted = false
        )
        dao.insertTask(task)
`
        // Verify task exists in active list
        val activeTasks = dao.getTasksByCategory(categoryId).first()
        assertEquals(1, activeTasks.size)
        assertEquals("Write QA Plan", activeTasks[0].title)
    }
`
    @Test
    fun completeTaskAndCheckCompletedList() = runBlocking {
        val categoryId = dao.insertCategory(CategoryEntity(name = "Personal"))
        val taskId = dao.insertTask(TaskEntity(title = "Walk dog", categoryId = categoryId))
`
        // Get task from db and update complete status
        val task = TaskEntity(id = taskId, title = "Walk dog", isCompleted = true, categoryId = categoryId)
        dao.updateTask(task)
`
        // Should not be in active category tasks
        val active = dao.getTasksByCategory(categoryId).first()
        assertEquals(0, active.size)
`
        // Should be in completed tasks
        val completed = dao.getCompletedTasks().first()
        assertEquals(1, completed.size)
        assertEquals("Walk dog", completed[0].title)
    }
}
```

## 2. Manual Verification Checklist
`
| Phase | Test Scenario | Expected Result | Pass/Fail |
| :--- | :--- | :--- | :--- |
| **UX** | Tap Quick-Add input box | Keyboard opens immediately (< 0.2s); input gets focused. | |
| **UX** | Type task and hit Enter | Task is saved, appended to top, keyboard stays open for quick follow-ups. | |
| **Gestures** | Swipe task card 50% right | Card animates, green check background shows, card disappears, task is marked complete. | |
| **Gestures** | Swipe task card 50% left | Card animates, red trash background shows, task is deleted. | |
| **Monetization** | Swipe task to complete/delete | Task acts immediately; **Zero interstitial ads are triggered** during gestures. | |
| **Monetization** | Exit Settings back to main list | Interstitial ad triggers **only if 180s cooldown timer** has elapsed. | |
| **System** | Process death / App close | Upon reload, all tasks and current category state persist correctly. | |
