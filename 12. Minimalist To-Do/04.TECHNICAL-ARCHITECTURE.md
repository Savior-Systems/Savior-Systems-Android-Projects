## 2. Core Architectural Flow
The app follows a unidirectional data flow (UDF) pattern:
`
```
┌──────────────────┐      Flow     ┌──────────────────┐
│  Room Database   │ ------------> │ TaskRepository   │
└──────────────────┘               └──────────────────┘
         ▲                                   │
         │ Save / Update                     │ Flow<List<Task>>
         │                                   ▼
┌──────────────────┐  UiState      ┌──────────────────┐
│  Compose View    │ <------------ │  MainViewModel   │
└──────────────────┘               └──────────────────┘
```

## 3. SwipeToDismissBox Jetpack Compose Code Example
Below is the clean production-ready integration of Compose Material 3 `SwipeToDismissBox` to implement swiping gestures for completion and deletion:
`
```kotlin
package com.saviorsystems.minimalisttodo.ui.main.components
`
import androidx.compose.animation.animateColorAsState
import androidx.compose.foundation.background
import androidx.compose.foundation.layout.*
import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.Check
import androidx.compose.material.icons.filled.Delete
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.unit.dp
import com.saviorsystems.minimalisttodo.data.database.entities.TaskEntity
`
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun DismissibleTaskItem(
    task: TaskEntity,
    onComplete: (TaskEntity) -> Unit,
    onDelete: (TaskEntity) -> Unit,
    modifier: Modifier = Modifier,
    content: @Composable () -> Unit
) {
    val dismissState = rememberSwipeToDismissBoxState(
        confirmValueChange = { value ->
            when (value) {
                SwipeToDismissBoxValue.StartToEnd -> {
                    onComplete(task)
                    true
                }
                SwipeToDismissBoxValue.EndToStart -> {
                    onDelete(task)
                    true
                }
                SwipeToDismissBoxValue.Settled -> false
            }
        }
    )
`
    SwipeToDismissBox(
        state = dismissState,
        backgroundContent = {
            val color by animateColorAsState(
                targetValue = when (dismissState.targetValue) {
                    SwipeToDismissBoxValue.StartToEnd -> Color(0xFF2E7D32) // Forest Green (Success)
                    SwipeToDismissBoxValue.EndToStart -> Color(0xFFC62828) // Crimson Red (Delete)
                    SwipeToDismissBoxValue.Settled -> Color.Transparent
                },
                label = "DismissBackground"
            )
`
            Box(
                modifier = Modifier
                    .fillMaxSize()
                    .background(color)
                    .padding(horizontal = 20.dp),
                contentAlignment = when (dismissState.targetValue) {
                    SwipeToDismissBoxValue.StartToEnd -> Alignment.CenterStart
                    SwipeToDismissBoxValue.EndToStart -> Alignment.CenterEnd
                    else -> Alignment.CenterStart
                }
            ) {
                when (dismissState.targetValue) {
                    SwipeToDismissBoxValue.StartToEnd -> {
                        Icon(
                            imageVector = Icons.Default.Check,
                            contentDescription = "Complete Task",
                            tint = Color.White
                        )
                    }
                    SwipeToDismissBoxValue.EndToStart -> {
                        Icon(
                            imageVector = Icons.Default.Delete,
                            contentDescription = "Delete Task",
                            tint = Color.White
                        )
                    }
                    else -> {}
                }
            }
        },
        content = {
            Card(
                modifier = modifier.fillMaxWidth(),
                colors = CardDefaults.cardColors(
                    containerColor = MaterialTheme.colorScheme.surfaceVariant
                )
            ) {
                content()
            }
        }
    )
}
```

## 4. ViewModel & UI State Model
The ViewModel handles task insertion, swipe dismissal, category switching, and database emissions:
`
```kotlin
package com.saviorsystems.minimalisttodo.ui.main
`
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.saviorsystems.minimalisttodo.data.database.entities.TaskEntity
import com.saviorsystems.minimalisttodo.data.database.entities.CategoryEntity
import com.saviorsystems.minimalisttodo.data.repository.TaskRepository
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
`
data class MainUiState(
    val tasks: List<TaskEntity> = emptyList(),
    val categories: List<CategoryEntity> = emptyList(),
    val selectedCategoryId: Long = 0L, // 0L represents "All"
    val isLoading: Boolean = true
)
`
class MainViewModel(
    private val repository: TaskRepository
) : ViewModel() {
`
    private val _selectedCategoryId = MutableStateFlow(0L)
`
    val uiState: StateFlow<MainUiState> = combine(
        repository.getAllCategories(),
        _selectedCategoryId.flatMapLatest { catId ->
            if (catId == 0L) repository.getAllActiveTasks()
            else repository.getTasksByCategory(catId)
        },
        _selectedCategoryId
    ) { categories, tasks, selectedId ->
        MainUiState(
            tasks = tasks,
            categories = categories,
            selectedCategoryId = selectedId,
            isLoading = false
        )
    }.stateIn(
        scope = viewModelScope,
        started = SharingStarted.WhileSubscribed(5000),
        initialValue = MainUiState()
    )
`
    fun selectCategory(categoryId: Long) {
        _selectedCategoryId.value = categoryId
    }
`
    fun addTask(title: String, categoryId: Long) {
        viewModelScope.launch {
            val task = TaskEntity(
                title = title,
                categoryId = categoryId,
                isCompleted = false,
                createdAt = System.currentTimeMillis()
            )
            repository.insertTask(task)
        }
    }
`
    fun completeTask(task: TaskEntity) {
        viewModelScope.launch {
            repository.updateTask(task.copy(isCompleted = true))
        }
    }
`
    fun deleteTask(task: TaskEntity) {
        viewModelScope.launch {
            repository.deleteTask(task)
        }
    }
}
```
